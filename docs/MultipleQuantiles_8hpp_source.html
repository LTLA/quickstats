39 MultipleQuantilesFixedNumber(const Number_ num_obs, const Quantiles_& quantiles) :
40 my_len(num_obs),
41 my_stacks(sanisizer::cast<I<decltype(my_stacks.size())> >(quantiles.size()))
42 {
43 static_assert(std::is_integral<Number_>::value);
44 static_assert(std::is_floating_point<Output_>::value);
45
46 if (num_obs <= 0) {
47 throw std::runtime_error("'num_obs' should be positive");
48 }
49 const Number_ num_m1 = num_obs - 1;
50
51 const auto num_quantiles = quantiles.size();
52 for (I<decltype(num_quantiles)> q = 0; q < num_quantiles; ++q) {
53 const auto quantile = quantiles[q];
54 if (q && quantile < quantiles[q - 1]) {
55 throw std::runtime_error("elements of 'quantiles' should be sorted");
56 }
57
58 auto& current = my_stacks[q];
59 configure_single_quantile<Output_, Number_>(quantile, num_m1, current.upper_index, current.upper_fraction, current.skip_lower);
60 }

86 template<typename Input_, class OutputFun_>
87 void operator()(Input_* const ptr, OutputFun_ output) const {
88 const auto end = ptr + my_len;
89 const auto num_quantiles = my_stacks.size();
90
91 Number_ last_index;
92 Output_ lower_val;
93
94 for (I<decltype(num_quantiles)> q = 0; q < num_quantiles; ++q) {
95 const auto& curstack = my_stacks[q];
96 const auto curindex = curstack.upper_index;
97
98 const auto target = ptr + curindex;
99 if (q == 0) {
100 std::nth_element(ptr, target, end);
101 } else if (curindex > last_index) {
102 // +1, as we only need to search after the last_index; everything before or at last_index is known to be too small.
103 std::nth_element(ptr + last_index + 1, target, end);
104 }
105
106 const Output_ upper_val = *target;
107 if (curstack.skip_lower) {
108 output(q, upper_val);
109 } else {
110 if (q == 0) {
111 lower_val = *std::max_element(ptr, target);
112 } else if (curindex > last_index) {
113 // No + 1, as ptr[last_index] might be the maximum.
114 lower_val = *std::max_element(ptr + last_index, target);
115 }
116 output(q, lower_val + (upper_val - lower_val) * curstack.upper_fraction);
117 }
118
119 last_index = curindex;
120 }
121 }

139 template<typename Input_, class OutputFun_>
140 void operator()(const Number_ nnz, Input_* const ptr, OutputFun_ output) const {
141 assert(nnz <= my_len);
142 const auto num_quantiles = my_stacks.size();
143
144 if (nnz == my_len) {
145 operator()(ptr, std::move(output));
146 return;
147 } else if (nnz == 0) {
148 for (I<decltype(num_quantiles)> q = 0; q < num_quantiles; ++q) {
149 output(q, 0);
150 }
151 return;
152 }
153
154 Number_ num_negative = 0;
155 for (Number_ i = 0; i < nnz; ++i) {
156 num_negative += (ptr[i] < 0);
157 }
158
159 // We need to use an optional here, as last_index may not be initialized for q > 0, e.g., if many quantiles are zero.
160 std::optional<Number_> last_index;
161 Output_ lower_val;
162 I<decltype(num_quantiles)> q = 0;
163
164 // Grind through all of the quantiles corresponding to negative values.
165 for (; q < num_quantiles; ++q) {
166 const auto& curstack = my_stacks[q];
167 const auto curindex = curstack.upper_index;
168 if (curindex >= num_negative) {
169 break;
170 }
171
172 const auto target = ptr + curindex;
173 if (!last_index.has_value()) {
174 std::nth_element(ptr, target, ptr + nnz);
175 } else if (curindex > *last_index) {
176 std::nth_element(ptr + *last_index + 1, target, ptr + nnz);
177 }
178
179 const Output_ upper_val = *target;
180 if (curstack.skip_lower) {
181 output(q, upper_val);
182 } else {
183 if (!last_index.has_value()) {
184 lower_val = *std::max_element(ptr, target);
185 } else if (curindex > *last_index) {
186 lower_val = *std::max_element(ptr + *last_index, target);
187 }
188 output(q, lower_val + (upper_val - lower_val) * curstack.upper_fraction);
189 }
190
191 last_index = curindex;
192 }
193
194 if (num_negative) {
195 std::optional<Output_> upper_val;
196
197 for (; q < num_quantiles; ++q) {
198 const auto& curstack = my_stacks[q];
199 const auto curindex = curstack.upper_index;
200 if (curindex > num_negative) {
201 break;
202 }
203
204 if (curstack.skip_lower) {
205 // Upper value must be zero.
206 // It can't be positive, as that would imply that there are no structural zeros and nnz == my_len.
207 output(q, 0);
208
209 } else {
210 if (!upper_val.has_value()) { // Only need to find it once given that everyone in this loop has 'upper_index == num_negative'.
211 const auto num_negative_m1 = num_negative - 1;
212 const auto target = ptr + num_negative_m1;
213
214 if (!last_index.has_value()) {
215 std::nth_element(ptr, target, ptr + nnz);
216 } else if (num_negative_m1 > *last_index) {
217 // Don't add 1, as we only know that last_index < num_negative from the break condition in the previous loop.
218 // It could be possible that last_index == num_negative - 1, in which case adding 1 would cause the start to go past 'target'.
219 std::nth_element(ptr + *last_index, target, ptr + nnz);
220 }
221
222 last_index = num_negative_m1;
223 upper_val = *target;
224 }
225
226 output(q, *upper_val * (1 - curstack.upper_fraction));
227 }
228 }
229 }
230
231 // Now we grind through all of the quantiles corresponding to zero values.
232 const Number_ num_zeros = my_len - nnz;
233 const Number_ num_not_positive = num_zeros + num_negative;
234 for (; q < num_quantiles; ++q) {
235 if (my_stacks[q].upper_index >= num_not_positive) {
236 break;
237 }
238 output(q, 0);
239 }
240
241 // Finally, we grind through the quantiles corresponding to positive values.
242 for (; q < num_quantiles; ++q) {
243 const auto& curstack = my_stacks[q];
244 const Number_ curindex = curstack.upper_index - num_zeros;
245
246 const auto target = ptr + curindex;
247 if (!last_index.has_value()) {
248 std::nth_element(ptr, target, ptr + nnz);
249 } else if (curindex > *last_index) {
250 // Even with subtraction of num_zeros, we can be sure that curindex > last_index at the start of this loop.
251 // This is because last_index must be pointing to a negative value if q > 0 at the start of this loop.
252 std::nth_element(ptr + *last_index + 1, target, ptr + nnz);
253 }
254
255 const Output_ upper_val = *target;
256 if (curstack.skip_lower) {
257 output(q, upper_val);
258 } else if (curindex == num_negative) {
259 output(q, upper_val * curstack.upper_fraction);
260 } else {
261 if (!last_index.has_value()) {
262 lower_val = *std::max_element(ptr, target);
263 } else if (curindex > *last_index) {
264 lower_val = *std::max_element(ptr + *last_index, target);
265 }
266 output(q, lower_val + (upper_val - lower_val) * curstack.upper_fraction);
267 }
268
269 last_index = curindex;
270 }
271 }
