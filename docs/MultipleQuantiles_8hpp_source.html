98 void operator()(Input_* const ptr, OutputFun_ output) const {
99 const auto end = ptr + my_len;
100 const auto num_quantiles = my_stacks.size();
101
102 std::size_t last_index = 0;
103 std::size_t last_index_p1 = 0; // yes, this is a deliberate starting value for 'last_index + 1'.
104 Output_ lower_val, upper_val;
105
106 for (I<decltype(num_quantiles)> q = 0; q < num_quantiles; ++q) {
107 const auto& curstack = my_stacks[q];
108 const auto curindex = curstack.upper_index;
109
110 // Only need to search if 'cur_index > last_index' (i.e., 'cur_index >= last_index_p1'),
111 // otherwise 'cur_index == last_index' and we can re-use existing lower/upper_val.
112 // The exception is if at the first iteration of this loop where 'last_index_p1 == 0 <= curindex',
113 // where this condition is always true to force a search to initialize 'upper_val'.
114 if (curindex >= last_index_p1) {
115 const auto target = ptr + curindex;
116
117 // +1, as we only need to search after the 'last_index'; everything before or at 'last_index' is known to be too small.
118 // Again, the exception is at the first iteration of this loop, where we need to search from the start of the array to initialize 'upper_val'.
119 std::nth_element(ptr + last_index_p1, target, end);
120 upper_val = *target;
121
122 if (curstack.skip_lower) {
123 output(q, upper_val);
124 } else {
125 // No + 1, as 'ptr[last_index]' might be the maximum.
126 lower_val = *std::max_element(ptr + last_index, target);
127 output(q, interpolate(lower_val, upper_val, curstack.upper_fraction));
128 }
129
130 last_index = curindex;
131 last_index_p1 = curindex + 1;
132
133 } else {
134 if (curstack.skip_lower) {
135 output(q, upper_val);
136 } else {
137 // It is implicitly guaranteed that 'lower_val' is initialized at this point.
138 // It's impossible for an earlier quantile to have 'skip_lower = true' while a later quantile has 'skip_lower = false'.
139 output(q, interpolate(lower_val, upper_val, curstack.upper_fraction));
140 }
141 }

164 void operator()(const std::size_t num_non_zero, Input_* const values, OutputFun_ output) const {
165 assert(num_non_zero <= my_len);
166 const auto num_quantiles = my_stacks.size();
167
168 if (num_non_zero == my_len) {
169 operator()(values, std::move(output));
170 return;
171 } else if (num_non_zero == 0) {
172 for (I<decltype(num_quantiles)> q = 0; q < num_quantiles; ++q) {
173 output(q, 0);
174 }
175 return;
176 }
177
178 std::size_t num_negative = 0;
179 for (std::size_t i = 0; i < num_non_zero; ++i) {
180 num_negative += (values[i] < 0);
181 }
182
183 std::size_t last_index = 0;
184 std::size_t last_index_p1 = 0; // see comments for the dense case for an explanation.
185 I<decltype(num_quantiles)> q = 0;
186
187 // Processing all quantiles where both upper and lower values are negative.
188 {
189 Output_ lower_val, upper_val;
190 for (; q < num_quantiles; ++q) {
191 const auto& curstack = my_stacks[q];
192 const auto curindex = curstack.upper_index;
193 if (curindex >= num_negative) {
194 break;
195 }
196
197 if (curindex >= last_index_p1) {
198 const auto target = values + curindex;
199 std::nth_element(values + last_index_p1, target, values + num_non_zero);
200 upper_val = *target;
201
202 if (curstack.skip_lower) {
203 output(q, upper_val);
204 } else {
205 lower_val = *std::max_element(values + last_index, target);
206 output(q, interpolate(lower_val, upper_val, curstack.upper_fraction));
207 }
208
209 last_index = curindex;
210 last_index_p1 = curindex + 1;
211
212 } else {
213 if (curstack.skip_lower) {
214 output(q, upper_val);
215 } else {
216 output(q, interpolate(lower_val, upper_val, curstack.upper_fraction));
217 }
218 }
219 }
220 }
221
222 // Processing all quantiles where lower value is negative and upper value is zero.
223 if (num_negative) {
224 bool computed = false;
225 Output_ lower_val;
226 for (; q < num_quantiles; ++q) {
227 const auto& curstack = my_stacks[q];
228 const auto curindex = curstack.upper_index;
229 if (curindex > num_negative) {
230 break;
231 }
232
233 if (curstack.skip_lower) {
234 // Upper value must be zero.
235 // It can't be positive, as that would imply that there are no structural zeros and 'num_non_zero == my_len'.
236 output(q, 0);
237 continue;
238 }
239
240 if (!computed) { // Only need to search this once given that everyone in this loop has 'upper_index == num_negative'.
241 const auto num_negative_m1 = num_negative - 1;
242 const auto target = values + num_negative_m1;
243
244 // We know that 'curindex >= num_negative' from the exit condition in the previous loop.
245 // This implies that any previous loop iterations would have 'curindex < num_negative' such that 'last_index + 1 <= num_negative'.
246 // If 'last_index + 1 == num_negative', then 'last_index == num_negative - 1', and no search is required.
247 // So, we only search if 'last_index + 1 < num_negative', i.e., 'last_index + 1 <= num_negative - 1'.
248 // (If there were no runs of the previous loop, then 'last_index + 1 == 0' and we force a search anyway.)
249 if (num_negative_m1 >= last_index_p1) {
250 std::nth_element(values + last_index_p1, target, values + num_non_zero);
251 }
252 lower_val = *target;
253
254 last_index = num_negative_m1;
255 last_index_p1 = num_negative;
256 computed = true;
257 }
258
259 output(q, lower_val * (1 - curstack.upper_fraction));
260 }
261 }
262
263 // Processing all quantiles where both the lower and upper values are zero.
264 const std::size_t num_zeros = my_len - num_non_zero;
265 const std::size_t num_not_positive = num_zeros + num_negative;
266 for (; q < num_quantiles; ++q) {
267 if (my_stacks[q].upper_index >= num_not_positive) {
268 break;
269 }
270 output(q, 0);
271 }
272
273 // Processing all quantiles where the lower value is zero and the upper value is positive.
274 {
275 bool computed = false;
276 Output_ upper_val;
277 for (; q < num_quantiles; ++q) {
278 const auto& curstack = my_stacks[q];
279 if (curstack.upper_index > num_not_positive) {
280 break;
281 }
282
283 if (!computed) { // Only need to search this once given that everyone in this loop has 'upper_index == num_not_positive'.
284 // The actual target is the first positive value at 'values[num_not_positive - num_zeros]', i.e., 'values[num_negative]'.
285 const auto target = values + num_negative;
286
287 // No point wrapping this in 'num_negative >= last_index + 1', as this will always be true from the exit conditions above.
288 // Or if none of the loops above were iterated over, we would still have 'last_index_p1 == 0', in which case this will also be true.
289 std::nth_element(values + last_index_p1, target, values + num_non_zero);
290
291 last_index = num_negative;
292 last_index_p1 = num_negative + 1;
293 upper_val = *target;
294 computed = true;
295 }
296
297 if (curstack.skip_lower) {
298 output(q, upper_val);
299 } else {
300 output(q, upper_val * curstack.upper_fraction);
301 }
302 }
303 }
304
305 // Processing all quantiles where the upper and lower values are positive.
306 {
307 Output_ upper_val, lower_val;
308 for (; q < num_quantiles; ++q) {
309 const auto& curstack = my_stacks[q];
310
311 // This should always be positive as we know that 'upper_index > num_not_positive' and thus 'upper_index - num_zeros > num_negative >= 0'.
312 const std::size_t curindex = curstack.upper_index - num_zeros;
313
314 // Even with subtraction of num_zeros, we can be sure that 'curindex >= last_index + 1' in the first iteration of this loop.
315 // From the clauses above, we know that 'last_index <= num_negative' and 'upper_index > num_not_positive'.
316 // So, subtracting 'num_zeros' gives us 'curindex > last_index' and thus 'curindex >= last_index + 1'.
317 // This ensures that this condition will always be true, and the code will be run, and 'upper_val' will always be set before use.
318 if (curindex >= last_index_p1) {
319 const auto target = values + curindex;
320 std::nth_element(values + last_index_p1, target, values + num_non_zero);
321 upper_val = *target;
322

324 output(q, upper_val);
