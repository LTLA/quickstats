36 SingleQuantileFixedNumber(const Number_ num, const Output_ quantile) : my_num(num) {
37 static_assert(std::is_integral<Number_>::value);
38 static_assert(std::is_floating_point<Output_>::value);
39
40 if (num <= 0) {
41 throw std::out_of_range("'num' should be positive");
42 }
43 if (quantile < 0 || quantile > 1) {
44 throw std::out_of_range("'quantile' should lie in [0, 1]");
45 }
46
47 const Number_ num_m1 = num - 1;
48 const Output_ raw = static_cast<Output_>(num_m1) * quantile;
49 const Output_ upper_index = std::ceil(raw);
50 const Output_ lower_index = std::floor(raw);
51
52 // Protect the cast from conversion imprecision between size_t and Float_,
53 // e.g., if it rounds up or if it converts it to an Inf.
54 const auto upper_san = sanisizer::from_float<Number_>(upper_index);
55 if (upper_san <= num_m1) {
56 my_upper_index = upper_san;
57 my_upper_fraction = raw - lower_index;
58 my_skip_lower = (upper_index == lower_index);
59 } else {
60 // Just gently cap it at the maximum value.
61 my_upper_index = num_m1;
62 my_upper_fraction = 0;
63 my_skip_lower = true;
64 }
65 }
66
67private:
68 Number_ my_num, my_upper_index;
69 Output_ my_upper_fraction;
70 bool my_skip_lower;
71
72public:
82 template<typename Input_>
83 Output_ operator()(Input_* ptr) const {
84 const auto target = ptr + my_upper_index;
85 std::nth_element(ptr, target, ptr + my_num);
86
87 // Avoid an extra memory access to get the lower index if we don't need it; this would also fail if upper_index = 0.
88 const Output_ upper = *target;
89 if (my_skip_lower) {
90 return upper;
91 }
92
93 const Output_ lower = *std::max_element(ptr, target);
94 return lower + (upper - lower) * my_upper_fraction;
95 }

111 template<typename Input_>
112 Output_ operator()(const Number_ nnz, Input_* ptr) const {
113 assert(nnz <= my_num);
114 if (nnz == my_num) {
115 return operator()(ptr);
116 } else if (nnz == 0) {
117 return 0;
118 }
119
120 Number_ nnz_negative = 0;
121 for (Number_ i = 0; i < nnz; ++i) {
122 nnz_negative += (ptr[i] < 0);
123 }
124
125 if (my_upper_index < nnz_negative) {
126 const auto target = ptr + my_upper_index;
127 std::nth_element(ptr, target, ptr + nnz);
128
129 const Output_ upper = *target;
130 if (my_skip_lower) {
131 return upper;
132 }
133
134 const Output_ lower = *std::max_element(ptr, target);
135 return lower + (upper - lower) * my_upper_fraction;
136 }
137
138 if (nnz_negative && my_upper_index == nnz_negative) {
139 // The upper value is zero. It can't be positive, as that would
140 // imply that there are no structural zeros and nnz == my_num.
141 if (my_skip_lower) {
142 return 0;
143 }
144
145 const auto target = ptr + (my_upper_index - 1);
146 std::nth_element(ptr, target, ptr + nnz);
147 return static_cast<Output_>(*target) * (1 - my_upper_fraction);
148 }
149
150 const Number_ nnz_zeros = my_num - nnz;
151 const Number_ nnz_not_positive = nnz_zeros + nnz_negative;
152 if (my_upper_index < nnz_not_positive) {
153 return 0;
154 }
155
156 const auto target = ptr + (my_upper_index - nnz_zeros);
157 std::nth_element(ptr, target, ptr + nnz);
158 const Output_ upper = *target;
159 if (my_skip_lower) {
160 return upper;
161 }
162
163 if (sanisizer::is_equal(my_upper_index, nnz_not_positive)) {
164 return upper * my_upper_fraction;
165 }
166
167 const Output_ lower = *std::max_element(ptr, target);
168 return lower + (upper - lower) * my_upper_fraction;
