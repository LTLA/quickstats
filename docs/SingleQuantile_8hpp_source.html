23template<typename Output_, typename Number_>
24void configure_single_quantile(
25 const Output_ quantile,
26 const Number_ num_m1,
27 Number_& upper_index,
28 Output_& upper_fraction,
29 bool& skip_lower
30) {
31 if (quantile < 0 || quantile > 1) {
32 throw std::out_of_range("'quantile' should lie in [0, 1]");
33 }
34
35 const Output_ raw_index = static_cast<Output_>(num_m1) * quantile;
36 const Output_ raw_upper_index = std::ceil(raw_index);
37 const Output_ raw_lower_index = std::floor(raw_index);
38
39 // Protect the cast from conversion imprecision between size_t and Float_,
40 // e.g., if it rounds up or if it converts it to an Inf.
41 const auto converted_upper_index = sanisizer::from_float<Number_>(raw_upper_index);
42 if (converted_upper_index <= num_m1) {
43 upper_index = converted_upper_index;
44 upper_fraction = raw_index - raw_lower_index;
45 skip_lower = (raw_upper_index == raw_lower_index);
46 } else {
47 // Just gently cap it at the maximum value.
48 upper_index = num_m1;
49 upper_fraction = 0;
50 skip_lower = true;
51 }
52}
65template<typename Output_, typename Number_>
66class SingleQuantileFixedNumber {
67public:
73 SingleQuantileFixedNumber(const Number_ num, const Output_ quantile) : my_num(num) {
74 static_assert(std::is_integral<Number_>::value);
75 static_assert(std::is_floating_point<Output_>::value);
76
77 if (num <= 0) {
78 throw std::out_of_range("'num' should be positive");
79 }
80 const Number_ num_m1 = num - 1;
81
82 configure_single_quantile(quantile, num_m1, my_upper_index, my_upper_fraction, my_skip_lower);
83 }
84
85private:
86 Number_ my_num, my_upper_index;
87 Output_ my_upper_fraction;
88 bool my_skip_lower;
89
90public:
100 template<typename Input_>
101 Output_ operator()(Input_* ptr) const {
102 const auto target = ptr + my_upper_index;
103 std::nth_element(ptr, target, ptr + my_num);
104
105 // Avoid an extra memory access to get the lower index if we don't need it; this would also fail if upper_index = 0.
106 const Output_ upper = *target;
107 if (my_skip_lower) {
108 return upper;
109 }
110
111 const Output_ lower = *std::max_element(ptr, target);
112 return lower + (upper - lower) * my_upper_fraction;
113 }
114
129 template<typename Input_>
130 Output_ operator()(const Number_ nnz, Input_* ptr) const {
131 assert(nnz <= my_num);
132 if (nnz == my_num) {
133 return operator()(ptr);
134 } else if (nnz == 0) {
135 return 0;
136 }
137
138 Number_ num_negative = 0;
139 for (Number_ i = 0; i < nnz; ++i) {
140 num_negative += (ptr[i] < 0);
141 }
142
143 if (my_upper_index < num_negative) {
144 const auto target = ptr + my_upper_index;
145 std::nth_element(ptr, target, ptr + nnz);
146
147 const Output_ upper = *target;
148 if (my_skip_lower) {
149 return upper;
150 }
151
152 const Output_ lower = *std::max_element(ptr, target);
153 return lower + (upper - lower) * my_upper_fraction;
154 }
155
156 if (num_negative && my_upper_index == num_negative) {
157 // The upper value is zero. It can't be positive, as that would
158 // imply that there are no structural zeros and nnz == my_num.
159 if (my_skip_lower) {
160 return 0;
161 }
162
163 const auto target = ptr + (my_upper_index - 1);
164 std::nth_element(ptr, target, ptr + nnz);
165 return static_cast<Output_>(*target) * (1 - my_upper_fraction);
166 }
167
168 const Number_ num_zeros = my_num - nnz;
169 const Number_ num_not_positive = num_zeros + num_negative;
170 if (my_upper_index < num_not_positive) {
171 return 0;
172 }
173
174 const auto target = ptr + (my_upper_index - num_zeros);
175 std::nth_element(ptr, target, ptr + nnz);
176 const Output_ upper = *target;
177 if (my_skip_lower) {
178 return upper;
179 }
180
181 if (my_upper_index == num_not_positive) {
182 return upper * my_upper_fraction;
183 }
184
185 const Output_ lower = *std::max_element(ptr, target);
186 return lower + (upper - lower) * my_upper_fraction;
187 }
