36 SingleQuantileFixedNumber(const Number_ num, const Output_ quantile) : my_num(num) {
37 assert(num > 0);
38 static_assert(std::is_integral<Number_>::value);
39 static_assert(std::is_floating_point<Output_>::value);
40
41 if (quantile < 0 || quantile > 1) {
42 throw std::out_of_range("'quantile' should lie in [0, 1]");
43 }
44
45 const Number_ num_m1 = num - 1;
46 const Output_ raw = static_cast<Output_>(num_m1) * quantile;
47 const Output_ upper_index = std::ceil(raw);
48 const Output_ lower_index = std::floor(raw);
49
50 // Protect the cast from conversion imprecision between size_t and Float_,
51 // e.g., if it rounds up or if it converts it to an Inf.
52 const auto upper_san = sanisizer::from_float<Number_>(upper_index);
53 if (upper_san <= num_m1) {
54 my_upper_index = upper_san;
55 my_upper_fraction = raw - lower_index;
56 my_skip_lower = (upper_index == lower_index);
57 } else {
58 // Just gently cap it at the maximum value.
59 my_upper_index = num_m1;
60 my_upper_fraction = 0;
61 my_skip_lower = true;
62 }
63 }
64
65private:
66 Number_ my_num, my_upper_index;
67 Output_ my_upper_fraction;
68 bool my_skip_lower;
69
70public:
78 Output_ operator()(Input_* ptr) const {
79 const auto target = ptr + my_upper_index;
80 std::nth_element(ptr, target, ptr + my_num);
81
82 // Avoid an extra memory access to get the lower index if we don't need it; this would also fail if upper_index = 0.
83 const Output_ upper = *target;
84 if (my_skip_lower) {
85 return upper;
86 }
87
88 const Output_ lower = *std::max_element(ptr, target);
89 return lower + (upper - lower) * my_upper_fraction;
90 }
91
104 Output_ operator()(const Number_ nnz, Input_* ptr) const {
105 assert(nnz <= my_num);
106 if (nnz == my_num) {
107 return operator()(ptr);
108 } else if (nnz == 0) {
109 return 0;
110 }
111
112 Number_ nnz_negative = 0;
113 for (Number_ i = 0; i < nnz; ++i) {
114 nnz_negative += (ptr[i] < 0);
115 }
116
117 if (my_upper_index < nnz_negative) {
118 const auto target = ptr + my_upper_index;
119 std::nth_element(ptr, target, ptr + nnz);
120
121 const Output_ upper = *target;
122 if (my_skip_lower) {
123 return upper;
124 }
125
126 const Output_ lower = *std::max_element(ptr, target);
127 return lower + (upper - lower) * my_upper_fraction;
128 }
129
130 if (nnz_negative && my_upper_index == nnz_negative) {
131 // The upper value is zero. It can't be positive, as that would
132 // imply that there are no structural zeros and nnz == my_num.
133 if (my_skip_lower) {
134 return 0;
135 }
136
137 const auto target = ptr + (my_upper_index - 1);
138 std::nth_element(ptr, target, ptr + nnz);
139 return static_cast<Output_>(*target) * (1 - my_upper_fraction);
140 }
141
142 const Number_ nnz_zeros = my_num - nnz;
143 const Number_ nnz_not_positive = nnz_zeros + nnz_negative;
144 if (my_upper_index < nnz_not_positive) {
145 return 0;
146 }
147
148 const auto target = ptr + (my_upper_index - nnz_zeros);
149 std::nth_element(ptr, target, ptr + nnz);
150 const Output_ upper = *target;
151 if (my_skip_lower) {
152 return upper;
153 }
154
155 if (sanisizer::is_equal(my_upper_index, nnz_not_positive)) {
156 return upper * my_upper_fraction;
157 }
158
159 const Output_ lower = *std::max_element(ptr, target);
160 return lower + (upper - lower) * my_upper_fraction;
161 }
