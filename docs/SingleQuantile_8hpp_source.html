45 if (converted_upper_index <= num_m1) {

47 upper_fraction = raw_index - raw_lower_index;
48 skip_lower = (raw_upper_index == raw_lower_index);
49 } else {
50 // Just gently cap it at the maximum value.
51 upper_index = num_m1;
52 upper_fraction = 0;
