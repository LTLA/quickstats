61template<typename Output_ = double, typename Input_>
62RssResult<Output_> rss(const std::size_t num_total, const std::size_t num_non_zero, const Input_* const ptr) {
63 static_assert(std::is_floating_point<Output_>::value);
64
65 RssResult<Output_> output;
66 if (num_total == 0) {
67 output.mean = std::numeric_limits<Output_>::quiet_NaN();
68 return output;
69 }
70
71 Output_& mean = output.mean;
72 for (std::size_t i = 0; i < num_non_zero; ++i) {
73 mean += ptr[i];
74 }
75 mean /= num_total;
76
77 Output_& ssd = output.rss;
78 for (std::size_t i = 0; i < num_non_zero; ++i) {
79 const auto delta = static_cast<Output_>(ptr[i]) - mean;
80 ssd += delta * delta;
81 }
82
83 assert(num_non_zero <= num_total);
84 if (num_non_zero < num_total) {
85 ssd += static_cast<Output_>(num_total - num_non_zero) * mean * mean;
86 }
87
88 return output;
89}

117void welford_add(Output_& mean, Output_& sumsq, const Value_ value, const Count_ count) {
118 Output_ delta = static_cast<Output_>(value) - mean;
119 mean += delta / count;
