81template<typename Output_ = double, typename Input_>
82RssResult<Output_> rss(const std::size_t num_total, const std::size_t num_non_zero, const Input_* const ptr, RssWorkspace<Output_>& work) {
83 static_assert(std::is_floating_point<Output_>::value);
84
85 RssResult<Output_> output;
86 if (num_total == 0) {
87 output.mean = std::numeric_limits<Output_>::quiet_NaN();
88 return output;
89 }
90
91 Output_& mean = output.mean;
92 mean = pairwise_sum(num_non_zero, ptr, work.pswork);
93 mean /= num_total;
94
95 Output_& ssd = output.rss;
96 ssd = pairwise_sum(
97 num_non_zero,
98 ptr,
99 [&](std::size_t, const Input_ val) -> Output_ {
100 const auto delta = static_cast<Output_>(val) - mean;
101 return delta * delta;
102 },
103 work.pswork
104 );
105
106 assert(num_non_zero <= num_total);
107 if (num_non_zero < num_total) {
108 ssd += static_cast<Output_>(num_total - num_non_zero) * mean * mean;
109 }
110
111 return output;
112}

141void welford_add(Output_& mean, Output_& sumsq, const Value_ value, const Count_ count) {
142 Output_ delta = static_cast<Output_>(value) - mean;
143 mean += delta / count;
144 sumsq += delta * (static_cast<Output_>(value) - mean);
