81template<std::size_t limit_ = 128, typename Input_, typename Output_>
82RssResult<Output_> rss(const std::size_t num_total, const std::size_t num_non_zero, const Input_* const ptr, RssWorkspace<Output_>& work) {
83 static_assert(std::is_floating_point<Output_>::value);
84
85 RssResult<Output_> output;
86 if (num_total == 0) {
87 output.mean = std::numeric_limits<Output_>::quiet_NaN();
88 return output;
89 }
90
91 Output_& mean = output.mean;
92 mean = pairwise_sum<limit_>(num_non_zero, ptr, work.pswork);
93 mean /= num_total;
94
95 Output_& ssd = output.rss;
96 ssd = pairwise_sum<limit_>(
97 num_non_zero,
98 ptr,
99 [&](std::size_t, const Input_ val) -> Output_ {
100 const auto delta = static_cast<Output_>(val) - mean;
101 return delta * delta;
102 },
103 work.pswork
104 );
105
106 assert(num_non_zero <= num_total);
107 if (num_non_zero < num_total) {
108 ssd += static_cast<Output_>(num_total - num_non_zero) * mean * mean;
109 }
110
111 return output;
112}

141template<typename Output_ = double, typename Value_, typename Count_>
142void welford_add(Output_& mean, Output_& sumsq, const Value_ value, const Count_ count) {
143 Output_ delta = static_cast<Output_>(value) - mean;
144 mean += delta / count;
145 sumsq += delta * (static_cast<Output_>(value) - mean);

488 void add(const std::size_t num_non_zero_obs, const Input_* value, const Index_* index, Skip_ skip) {
489 static_assert(std::is_integral<Index_>::value);
490
491 // my_count is the upper bound of all my_num_non_zero, so no need to check individual increments.
492 my_count = sanisizer::sum<Count_>(my_count, 1);
493

495 const auto val = value[i];
496 const auto ri = index[i];
497 if (skip(ri, val)) {

625template<typename Float_>
626void rss_to_variance(const std::size_t num_obj, const std::size_t num_total, Float_* const rss) {
627 if (num_total < 1) {
628 std::fill_n(rss, num_obj, std::numeric_limits<Float_>::quiet_NaN());
629 } else {
630 // For consistency with the other overloads, we won't do the '* (1/denom)' trick.
631 // It shouldn't have much effect on throughput anyway as the bottleneck should be reading from memory.
632 for (std::size_t i = 0; i < num_obj; ++i) {
633 rss[i] /= num_total - 1;
634 }
635 }
636}
