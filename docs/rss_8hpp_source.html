58template<typename Output_ = double, typename Input_>
59RssResult<Output_> rss(const std::size_t num_total, const std::size_t num_non_zero, const Input_* const ptr) {
60 static_assert(std::is_floating_point<Output_>::value);
61
62 RssResult<Output_> output;
63 if (num_total == 0) {
64 output.mean = std::numeric_limits<Output_>::quiet_NaN();
65 return output;
66 }
67
68 Output_& mean = output.mean;
69 for (std::size_t i = 0; i < num_non_zero; ++i) {
70 mean += ptr[i];
71 }
72 mean /= num_total;
73
74 Output_& ssd = output.rss;
75 for (std::size_t i = 0; i < num_non_zero; ++i) {
76 const auto delta = static_cast<Output_>(ptr[i]) - mean;
77 ssd += delta * delta;
78 }
79
80 assert(num_non_zero <= num_total);
81 if (num_non_zero < num_total) {
82 ssd += static_cast<Output_>(num_total - num_non_zero) * mean * mean;
83 }
84
85 return output;
86}
