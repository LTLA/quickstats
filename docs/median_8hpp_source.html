31template<typename Output_, typename Number_, typename Input_>
32Output_ median(const Number_ num, Input_* const ptr) {
33 static_assert(std::is_integral<Number_>::value);
34 static_assert(std::is_floating_point<Output_>::value);
35
36 if (num == 0) {
37 return std::numeric_limits<Output_>::quiet_NaN();
38 }
39
40 const Number_ halfway = num / 2;
41 const bool is_even = (num % 2 == 0);
42
43 std::nth_element(ptr, ptr + halfway, ptr + num);
44 const Output_ medtmp = *(ptr + halfway);
45 if (!is_even) {
46 return medtmp;
47 }
48
49 // 'nth_element()' reorganizes 'ptr' so that everything below 'halfway' is
50 // less than or equal to 'ptr[halfway]', while everything above 'halfway'
51 // is greater than or equal to 'ptr[halfway]'. Thus, to get the element
52 // immediately before 'halfway' in the sort order, we just need to find the
53 // maximum from '[0, halfway)'.
54 const Output_ other = *std::max_element(ptr, ptr + halfway);
55
56 if (medtmp == other) {
57 return medtmp; // Preserve exactness, respect infinities of the same sign.
58 } else {
59 return medtmp + (other - medtmp) / 2; // Avoid FP overflow.
60 }
61}

82template<typename Output_ = double, typename Number_, typename Input_>
83Output_ median(const Number_ num, const Number_ nnz, Input_* const value) {
84 assert(num >= nnz);
85
86 // Fallback to the dense code if there are no structural zeros. This is not
87 // just for efficiency as the downstream averaging code assumes that there
88 // is at least one structural zero when considering its scenarios.
89 if (nnz == num) {
90 return median<Output_>(num, value);
91 }
92
93 // Is the number of non-zeros less than the number of zeros?
94 // If so, the median must be zero. Note that we calculate it
95 // in this way to avoid overflow from 'nnz * 2'.
96 if (nnz < num - nnz) {
97 return 0;
98 }
99
100 const Number_ halfway = num / 2;
101 const bool is_even = (num % 2 == 0);
102
103 const Number_ num_zero = num - nnz;
104 Number_ num_negative = 0;
105 for (Number_ i = 0; i < nnz; ++i) {
106 num_negative += (value[i] < 0);
107 }
108
109 if (!is_even) {
110 if (num_negative > halfway) {
111 std::nth_element(value, value + halfway, value + nnz);
112 return value[halfway];
113
114 } else if (halfway >= num_negative + num_zero) {
115 const Number_ skip_zeros = halfway - num_zero;
116 std::nth_element(value, value + skip_zeros, value + nnz);
117 return value[skip_zeros];
118
119 } else {
120 return 0;
121 }
122 }
123
124 Output_ baseline = 0, other = 0;
125 if (num_negative > halfway) { // both halves of the median are negative.
126 std::nth_element(value, value + halfway, value + nnz);
127 baseline = value[halfway];
128 other = *(std::max_element(value, value + halfway)); // max_element gets the sorted value at halfway - 1, see explanation for the dense case.
129
130 } else if (num_negative == halfway) { // the upper half is guaranteed to be zero.
131 const Number_ below_halfway = halfway - 1;
132 std::nth_element(value, value + below_halfway, value + nnz);
133 other = value[below_halfway]; // set to other so that addition/subtraction of a zero baseline has no effect on precision.
134
135 } else if (num_negative < halfway && num_negative + num_zero > halfway) { // both halves are zero, so zero is the median.
136 ;
137
138 } else if (num_negative + num_zero == halfway) { // the lower half is guaranteed to be zero.
139 const Number_ skip_zeros = halfway - num_zero;
140 std::nth_element(value, value + skip_zeros, value + nnz);
141 other = value[skip_zeros]; // set to other so that addition/subtraction of a zero baseline has no effect on precision.
142
143 } else { // both halves of the median are non-negative.
144 const Number_ skip_zeros = halfway - num_zero;
145 std::nth_element(value, value + skip_zeros, value + nnz);
146 baseline = value[skip_zeros];
147 other = *(std::max_element(value, value + skip_zeros)); // max_element gets the sorted value at skip_zeros - 1, see explanation for the dense case.
148 }
149
150 if (baseline == other) {
151 return baseline; // Preserve exactness, respect infinities of the same sign.
152 } else {
153 return baseline + (other - baseline) / 2; // Avoid FP overflow.
154 }
155}
